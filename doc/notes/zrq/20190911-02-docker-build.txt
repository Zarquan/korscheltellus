# -----------------------------------------------------
# Build our containers.
#[user@desktop]

    source "${HOME}/korscheltellus.settings"
    pushd "${KORSCHELTELLUS_CODE:?}"

        export buildtag=$(date '+%Y.%m.%d')

        source bin/util.sh
        dockerfiles ${buildtag}

        docker build \
            --tag korscheltellus/fedora:${buildtag:?} \
            docker/fedora

        docker build \
            --tag korscheltellus/ansible-client:${buildtag:?} \
            docker/ansible-client

    popd

# -----------------------------------------------------
# Login to the Docker registry.
#[user@desktop]

    docker login \
        --username $(secret docker.io.user) \
        --password $(secret docker.io.pass)

# -----------------------------------------------------
# Tag and push our containers.
#[user@desktop]

    docker tag  "korscheltellus/fedora:${buildtag:?}" "korscheltellus/fedora:latest"
    docker push "korscheltellus/fedora:${buildtag:?}"
    docker push "korscheltellus/fedora:latest"

    docker tag  "korscheltellus/ansible-client:${buildtag:?}" "korscheltellus/ansible-client:latest"
    docker push "korscheltellus/ansible-client:${buildtag:?}"
    docker push "korscheltellus/ansible-client:latest"
